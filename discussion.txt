import { CCard, CCardHeader, CCardBody, CTable, CBadge, CButton, CInputGroup, CFormInput, CForm } from '@coreui/vue';

      type TEXT DEFAULT 'lettre', -- Lettre, Email, Colis

      this.outgoingMails = response.data;
    },
    async createIncomingMail(mailData) {
      const response = await axios.post('http://localhost:3000/api/mails/incoming', {
        ...mailData,
        assigned_to: this.currentUser?.id,
      });
      this.fetchIncomingMails();
      return response.data;
    },
    async updateMail({ id, status, archived, assigned_to }) {
      await axios.put(`http://localhost:3000/api/mails/update/${id}`, { status, archived, assigned_to });
      if (status) {
        await axios.post('http://localhost:3000/api/actions', { mail_id: id, mail_type: 'incoming', action: status, user_id: this.currentUser?.id });
      }
      this.fetchIncomingMails();
    },
    async deleteMail(id) {
      await axios.delete(`http://localhost:3000/api/mails/incoming/${id}`);
      this.fetchIncomingMails();
    },
    async fetchUsers() {
      const response = await axios.get('http://localhost:3000/api/users');
      this.users = response.data;
    },
    async fetchActionsLog(mailId, mailType) {
      const response = await axios.get(`http://localhost:3000/api/actions?mail_id=${mailId}&mail_type=${mailType}`);
      this.actionsLog = response.data;
    },
  },
  getters: {
    getIncomingMails: (state) => state.incomingMails,
    getOutgoingMails: (state) => state.outgoingMails,
    getUsers: (state) => state.users,
    getActionsLog: (state) => state.actionsLog,
    hasRole: (state) => (role) => state.currentUser?.role === role,
  },

<script>
import { reactive, computed } from 'vue';
import { useCourrierStore } from './stores/courrier';
import { useRouter, useRoute } from 'vue-router';
import { CIcon, CButton, CDropdown, CDropdownToggle, CDropdownMenu, CDropdownItem, CDropdownDivider, CBadge } from '@coreui/vue';

export default {
  name: 'App',
  components: {
    CIcon,
    CButton,
    CDropdown,
    CDropdownToggle,
    CDropdownMenu,
    CDropdownItem,
    CDropdownDivider,
    CBadge,
  },
  setup() {
    const store = useCourrierStore();
    const router = useRouter();
    const route = useRoute();
    const groups = reactive({
      entrants: true,
      sortants: true,
      admin: true,
    });
    const sidebarVisible = ref(true);

    const toggleGroup = (group) => {
      groups[group] = !groups[group];
    };

    const toggleSidebar = () => {
      sidebarVisible.value = !sidebarVisible.value;
    };

    const logout = () => {
      store.logout();
      router.push('/login');
    };

    const toggleTheme = (theme) => {
      document.body.setAttribute('data-theme', theme);
    };

    store.fetchRecentActions();
    const recentActions = computed(() => store.getRecentActions || []);

    const clearNotifications = () => {
      store.clearRecentActions();
    };

    const currentRouteName = computed(() => route.name || 'Dashboard');

    return {
      groups,
      toggleGroup,
      toggleSidebar,
      sidebarVisible,
      logout,
      recentActions,
      clearNotifications,
      toggleTheme,
      currentRouteName,
    };
  },
};
</script>

L’erreur SQLITE_ERROR: duplicate column name: archived_date signifie que la colonne archived_date existe déjà dans la table (probablement ajoutée lors d’une exécution précédente), et SQLite ne permet pas de l’ajouter à nouveau avec une commande ALTER TABLE. Pour résoudre ce problème, nous devons gérer cette migration de manière plus robuste en vérifiant si les colonnes existent avant d’essayer de les ajouter. Voici comment corriger cela :

const cors = require('cors');
const sqlite3 = require('sqlite3').verbose();
const app = express();

app.use(express.json({ strict: false }));
app.use((req, res, next) => {
  console.log(`[${req.method}] ${req.url} - Body:`, req.body);
  next();

const db = new sqlite3.Database('./mails.db', (err) => {
  if (err) console.error('Erreur de connexion à la base :', err);
  else console.log('Connecté à la base SQLite');
});

// Fonction pour vérifier si une colonne existe dans une table
const columnExists = (table, column) => {
  return new Promise((resolve) => {
    db.all(`PRAGMA table_info(${table})`, (err, rows) => {
      if (err) return resolve(false);
      resolve(rows.some(row => row.name === column));
    });
  });
};

// Migration conditionnelle pour ajouter les colonnes si elles n'existent pas
db.serialize(async () => {
  const tables = ['incoming_mails', 'outgoing_mails'];
  const columnsToAdd = [
    { name: 'archived_date', type: 'TEXT' },
    { name: 'archived_reason', type: 'TEXT' },
    { name: 'priority', type: 'TEXT DEFAULT "normal"' },
    { name: 'type', type: 'TEXT DEFAULT "lettre"' },
  ];

  for (const table of tables) {
    for (const col of columnsToAdd) {
      const exists = await columnExists(table, col.name);
      if (!exists) {
        db.run(`ALTER TABLE ${table} ADD COLUMN ${col.name} ${col.type}`, (err) => {
          if (err) console.error(`Erreur lors de l'ajout de ${col.name} à ${table} :`, err);
          else console.log(`Colonne ${col.name} ajoutée à ${table}`);
        });
      } else {
        console.log(`Colonne ${col.name} existe déjà dans ${table}`);
      }
    }
  }
});

const express = require('express');
const cors = require('cors');
const sqlite3 = require('sqlite3').verbose();
const app = express();

app.use(cors());

  console.log(`[${req.method}] ${req.url} - Body:`, req.body);
  next();

const db = new sqlite3.Database('./mails.db', (err) => {
  if (err) console.error('Erreur de connexion à la base :', err);
  else console.log('Connecté à la base SQLite');
});

// Vérification et ajout conditionnel des colonnes
const columnExists = (table, column) => {
  return new Promise((resolve) => {
    db.all(`PRAGMA table_info(${table})`, (err, rows) => {
      if (err) return resolve(false);
      resolve(rows.some(row => row.name === column));
    });
  });
};

db.serialize(async () => {
  const tables = ['incoming_mails', 'outgoing_mails'];
  const columnsToAdd = [
    { name: 'archived_date', type: 'TEXT' },
    { name: 'archived_reason', type: 'TEXT' },
    { name: 'priority', type: 'TEXT DEFAULT "normal"' },
    { name: 'type', type: 'TEXT DEFAULT "lettre"' },
  ];

  for (const table of tables) {
    for (const col of columnsToAdd) {
      const exists = await columnExists(table, col.name);
      if (!exists) {
        db.run(`ALTER TABLE ${table} ADD COLUMN ${col.name} ${col.type}`, (err) => {
          if (err) console.error(`Erreur lors de l'ajout de ${col.name} à ${table} :`, err);
          else console.log(`Colonne ${col.name} ajoutée à ${table}`);
        });
      } else {
        console.log(`Colonne ${col.name} existe déjà dans ${table}`);
      }
    }
  }

  // Création de la table users si elle n'existe pas
  db.run(`
    CREATE TABLE IF NOT EXISTS users (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      username TEXT NOT NULL UNIQUE,
      password TEXT NOT NULL,
      name TEXT NOT NULL
    )
  `);

  // Ajout d’un utilisateur par défaut si la table est vide
  db.get('SELECT COUNT(*) as count FROM users', (err, row) => {
    if (row.count === 0) {
      db.run('INSERT INTO users (username, password, name) VALUES (?, ?, ?)', ['admin', 'admin123', 'Administrateur']);
      console.log('Utilisateur par défaut créé : admin/admin123');
    }
  });
});

// Route de connexion
app.post('/api/login', (req, res) => {
  const { username, password } = req.body;
  db.get('SELECT * FROM users WHERE username = ? AND password = ?', [username, password], (err, row) => {
    if (err) return res.status(500).json({ error: err.message });
    if (!row) return res.status(401).json({ error: 'Identifiants incorrects' });
    res.json({ id: row.id, username: row.username, name: row.name });

<template>
  <div class="login-container">
    <h2>Connexion</h2>
    <CForm @submit.prevent="login">
      <CInputGroup class="mb-3">
        <CFormInput v-model="credentials.username" placeholder="Nom d'utilisateur" required />
      </CInputGroup>
      <CInputGroup class="mb-3">
        <CFormInput v-model="credentials.password" type="password" placeholder="Mot de passe" required />
      </CInputGroup>
      <CButton color="primary" type="submit">Se connecter</CButton>
    </CForm>
    <p v-if="error" class="text-danger">{{ error }}</p>
  </div>
</template>

<script>
import { reactive, ref } from 'vue';
import { useCourrierStore } from '../stores/courrier';
import { useRouter } from 'vue-router';
import { CForm, CInputGroup, CFormInput, CButton } from '@coreui/vue';

export default {
  name: 'Login',
  components: { CForm, CInputGroup, CFormInput, CButton },
  setup() {
    const store = useCourrierStore();
    const router = useRouter();
    const credentials = reactive({ username: '', password: '' });
    const error = ref('');

    const login = async () => {
      try {
        const response = await fetch('http://localhost:3000/api/login', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(credentials),
        });
        if (!response.ok) throw new Error('Identifiants incorrects');
        const user = await response.json();
        store.setUser(user);
        router.push('/dashboard');
      } catch (err) {
        error.value = err.message;
      }
    };

    return { credentials, login, error };
  },
};
</script>

export const useCourrierStore = defineStore('courrier', {
  state: () => ({
    incomingMails: [],

    recentActions: [],
    currentUser: null,
  }),
  getters: {
